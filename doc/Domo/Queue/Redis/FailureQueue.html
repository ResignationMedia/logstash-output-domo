def reprocess_jobs!(stream_execution_id=nil)
  queue = job_queue
  each do |job|
    if stream_execution_id.nil? or job.execution_id != stream_execution_id
      job.part_num = nil
    end
    job.execution_id = stream_execution_id
    queue.add(job)
  end
  # Clear this queue out and return the new queue
  clear
  queue
end
