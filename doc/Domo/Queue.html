def each(&block)
  return to_enum(:each) unless block_given?
  until @client.llen(@queue_name) <= 0
    job = pop
    break if job.nil?
    yield job
  end
end

def pop
  job = @client.lpop(@queue_name)
  return if job.nil?
  Job.from_json!(job)
end
